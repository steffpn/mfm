// Build where clause with scope filtering
const where: Record<string, unknown> = {};
if (q) { where.OR = [songTitle contains, artistName contains, isrc equals] }
if (startDate || endDate) { where.startedAt = { gte, lte } }
if (stationId) { where.stationId = stationId }
if (currentUser.role === "STATION") { where.stationId = { in: stationScopes } }

const events = await prisma.airplayEvent.findMany({
  where,
  orderBy: { startedAt: "desc" },
  include: { station: { select: { name: true } } },
});
